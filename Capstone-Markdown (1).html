library(tidyverse)
library(colorRamps)
library(knitr)
sleep_day <- read_csv("sleepDay_merged.csv") %>% 
  separate(SleepDay, into =c("ActivityDate", "Time"), sep = (' ')) 

## The separate function is used to make the date and time formats the same. 

daily_activity <- read_csv("dailyActivity_merged.csv")

Slp_Act_Merged <- left_join(daily_activity, sleep_day, by = c("Id", "ActivityDate")) %>% 
  mutate(ActiveMinutes = VeryActiveMinutes + FairlyActiveMinutes + LightlyActiveMinutes) %>%  
  mutate(TotalMinutes = SedentaryMinutes + ActiveMinutes + TotalTimeInBed) %>%
  filter(SedentaryMinutes >100) %>% 
  filter(TotalSteps > 100) %>% 
  filter(TotalMinutes > 720)

## Above filtering done to eliminate outliers where data collection is missing or incomplete. 
## Last filter makes sure that the subject was collecting data for at least half of the day, 720 minutes.
